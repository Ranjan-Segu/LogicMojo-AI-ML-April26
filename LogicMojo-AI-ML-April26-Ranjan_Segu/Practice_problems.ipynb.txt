        "  count = 0\n",
        "  for num in nums_list:\n",
        "    # Prime numbers are greater than 1\n",
        "    if num > 1:\n",
        "      is_prime = True\n",
        "      # Check for factors from 2 up to the square root of num\n",
        "      for i in range(2, int(num**0.5) + 1):\n",
        "        if num % i == 0:\n",
        "          is_prime = False\n",
        "          break # Not a prime number, no need to check further\n",
        "      if is_prime:\n",
        "        count += 1\n",
        "  return count\n",
